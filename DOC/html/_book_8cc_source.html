156 void Book::findExpression(string query, set<int> &pos) {
157  // Trim spaces of the query
158  utils::trimString(query);
159  if (query.find_first_of('{') == string::npos and query.find_first_of('}') == string::npos) {
160  // Base case: Malformed expression (not found any word to find)
161  } else if (query.find_last_of('{') == 0 and query.find_first_of('}') == query.length() - 1
162  and query.find_first_of('&') == string::npos and query.find_first_of('|') == string::npos) {
163  query = query.substr(1, query.length() - 2);
164  if (query.find_first_of(" ") == string::npos) {
165  // Base case: One word, return lines where found
166  map < string, list < int > > ::const_iterator
167  it1 = lineDictionary.find(query);
168  if (it1 != lineDictionary.end()) {
169  list<int>::const_iterator it2 = it1->second.begin();
170  while (it2 != it1->second.end()) {
171  pos.insert(*it2);
172  it2++;
173  }
174  }
175  } else {
176  // Recursive case: More than one word, replace spaces with &
177  istringstream iss(query);
178  query.clear();
179  string word;
180  iss >> word;
181  query += "{" + word + "}";
182  while (iss >> word) {
183  query += " & {" + word + "}";
184  }
185  findExpression(query, pos);
186  }
187  } else {
188  // Find the first & operator outside parentheses
189  int posAnd = query.find_first_of('&');
190  while (posAnd != string::npos and utils::betweenPar(query, posAnd)) {
191  posAnd = query.find_first_of('&', posAnd + 1);
192  }
193  // If found call the recursivity on both sides of the expression
194  if (posAnd != string::npos) {
195  // Recursive case: Call recursion of both expressions and intersect them
196  set<int> set1, set2;
197  findExpression(query.substr(0, posAnd), set1);
198  findExpression(query.substr(posAnd + 1), set2);
199  set_intersection(set1.begin(), set1.end(), set2.begin(), set2.end(), inserter(pos, pos.begin()));
200  } else {
201  // If no outer & operators, find the first | operator outside parentheses
202  int posOr = query.find_first_of('|');
203  while (posOr != string::npos and utils::betweenPar(query, posOr)) {
204  posOr = query.find_first_of('|', posOr + 1);
205  }
206  if (posOr != string::npos) {
207  // Recursive case: Call recursion of both expressions and union them
208  set<int> set1, set2;
209  findExpression(query.substr(0, posOr), set1);
210  findExpression(query.substr(posOr + 1), set2);
211  set_union(set1.begin(), set1.end(), set2.begin(), set2.end(), inserter(pos, pos.begin()));
212  } else if (!utils::betweenPar(query, query.length())) {
213  // Recursive case: Call recursion of inner parentheses expression
214  findExpression(query.substr(1, query.length() - 2), pos);
